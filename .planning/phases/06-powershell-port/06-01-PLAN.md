**Key reader `_tui_read_key()`** sets two globals: `_tui_rk_result` (symbolic key name) and `_tui_rk_digit` (digit char when NUMBER). NEVER wrap in `$()`.

**Escape sequence parsing:**
- `\033[A` = UP, `\033[B` = DOWN, `\033[C` = RIGHT, `\033[D` = LEFT
- `\033[5~` = PGUP, `\033[6~` = PGDN
- `\033[H` or `\033[1~` = HOME, `\033[F` or `\033[4~` = END
- `\033[3~` = DELETE
- `\033` followed by nothing = ESC (with timeout via `TUI_KEY_TIMEOUT`)

**Single-select widget `tui_select()`:**
- Signature: `tui_select title subtitle item1 item2 ...` — items as positional args
- Sets global `TUI_RESULT` to the selected 0-based index on Enter, -1 on Esc/q cancel
- Renders full-screen with box border, title, items with reverse-video highlight on current
- Scroll when items > visible area: `↑more` at top, `↓more` at bottom
- Help footer: "↑↓ jk move  ↵ select  Esc/q cancel  ? help  Home End PgUp PgDn"
- PgUp/PgDn jump by page height, Home/End jump to first/last
- Number jump: multi-digit accumulator; typing digits jumps to matching index
- Fallback: `_tui_select_fallback()` — numbered text prompt when `_tui_use_tui=false`

**Section: PS Version Detection and ANSI VT Enable (per D-03, D-04, D-05)**

```powershell
# Detect PS version at module load time:
$Script:FluPsVersion = $PSVersionTable.PSVersion  # D-05

# Detect ANSI/VT support:
# PS 7+ supports ANSI natively. PS 5.1 requires VT processing mode enable.
# Use $Host.UI.SupportsVirtualTerminal (PS 5.1+ when VT enabled) or manual check.
$Script:FluAnsiSupported = $false
if ($Host.UI.SupportsVirtualTerminal) {
    $Script:FluAnsiSupported = $true
} else {
    # Attempt to enable VT processing on PS 5.1
    try {
        $MethodDefinitions = @'
[DllImport("kernel32.dll", SetLastError = $true)]
public static extern IntPtr GetStdHandle(int nStdHandle);
[DllImport("kernel32.dll", SetLastError = $true)]
public static extern bool GetConsoleMode(IntPtr hConsoleHandle, out uint lpMode);
[DllImport("kernel32.dll", SetLastError = $true)]
public static extern bool SetConsoleMode(IntPtr hConsoleHandle, uint dwMode);
'@
        $Kernel32 = Add-Type -MemberDefinition $MethodDefinitions -Name 'Kernel32' -Namespace 'Win32' -PassThru
        $hConsoleHandle = $Kernel32::GetStdHandle(-11)  # STD_OUTPUT_HANDLE
        $mode = 0
        if ($Kernel32::GetConsoleMode($hConsoleHandle, [ref]$mode)) {
            $ENABLE_VIRTUAL_TERMINAL_PROCESSING = 0x0004
            if ($Kernel32::SetConsoleMode($hConsoleHandle, $mode -bor $ENABLE_VIRTUAL_TERMINAL_PROCESSING)) {
                $Script:FluAnsiSupported = $true
            }
        }
    } catch {
        # VT not available — ASCII fallback mode
    }
}
# Also set [Console]::OutputEncoding for Unicode box drawing support
try { [Console]::OutputEncoding = [System.Text.Encoding]::UTF8 } catch {}
```

**Section: Terminal Init/Restore Functions**

```powershell
function Initialize-Tui {
    if (-not $Script:_tui_use_tui) { return }
    # Save cursor visibility state, hide cursor
    $Script:_tui_cursor_visible = [Console]::CursorVisible
    [Console]::CursorVisible = $false
    # Register cleanup on Ctrl-C
    $Script:_tui_trap_set = $true
    [Console]::TreatControlCAsInput = $true
}

function Restore-Tui {
    if ($Script:_tui_cursor_visible) {
        [Console]::CursorVisible = $true
    }
    if ($Script:_tui_trap_set) {
        [Console]::TreatControlCAsInput = $false
    }
    Clear-TuiScreen
    Remove-Variable -Name '_tui_trap_set' -Scope Script -ErrorAction SilentlyContinue
    Remove-Variable -Name '_tui_cursor_visible' -Scope Script -ErrorAction SilentlyContinue
}
```

**Section: Rendering Primitives**

```powershell
function Move-TuiCursor {
    param([int]$Row, [int]$Col)
    if ($Script:FluAnsiSupported) {
        Write-Host "$ESC[${Row};${Col}H" -NoNewline
    } else {
        [Console]::SetCursorPosition($Col - 1, $Row - 1)
    }
}

function Clear-TuiScreen {
    if ($Script:FluAnsiSupported) {
        Write-Host "$ESC[2J$ESC[H" -NoNewline
    } else {
        [Console]::Clear()
    }
}

function Write-TuiAt {
    # write-TuiAt -Row N -Col M -Text "string" — positioned output
    param([int]$Row, [int]$Col, [string]$Text)
    Move-TuiCursor -Row $Row -Col $Col
    Write-Host $Text -NoNewline
}
```

```powershell
function Write-TuiBox {
    # write-TuiBox -X $x -Y $y -Width $w -Height $h -Title "header text"
    # Draws a bordered box matching tui.sh _tui_draw_box() exact visual output:
    #  - Top border: TUI_BOX_TL + (TUI_BOX_H * (W-2)) + TUI_BOX_TR
    #  - If Title: render centered in title row
    #  - Vertical borders: TUI_BOX_V on each side for remaining rows
    #  - Bottom border: TUI_BOX_BL + (TUI_BOX_H * (W-2)) + TUI_BOX_BR
    param([int]$X, [int]$Y, [int]$Width, [int]$Height, [string]$Title)

    $innerW = $Width - 2
    $hLine = $Script:TUI_BOX_H * $innerW

    # Top border
    Move-TuiCursor -Row $Y -Col $X
    Write-Host "$($Script:TUI_BOX_TL)$hLine$($Script:TUI_BOX_TR)" -NoNewline

    # Title row (centered) if provided
    if ($Title) {
        Move-TuiCursor -Row $Y -Col $X
        Write-Host $Script:TUI_BOX_V -NoNewline
        # Strip ANSI for length calculation (rough: remove ESC sequences)
        $plainTitle = $Title -replace '\x1b\[[0-9;]*m', ''
        $titleLen = $plainTitle.Length
        $padLeft = [Math]::Max(0, [Math]::Floor(($innerW - $titleLen) / 2))
        $padRight = $innerW - $titleLen - $padLeft
        Write-Host (" " * $padLeft) -NoNewline
        Write-Host $Title -NoNewline
        Write-Host (" " * [Math]::Max(0, $padRight)) -NoNewline
        Write-Host $Script:TUI_BOX_V -NoNewline

        # Fill remaining rows with empty borders below title
        for ($r = $Y + 2; $r -lt $Y + $Height - 1; $r++) {
            Move-TuiCursor -Row $r -Col $X
            Write-Host $Script:TUI_BOX_V -NoNewline
            Write-Host (" " * $innerW) -NoNewline
            Write-Host $Script:TUI_BOX_V -NoNewline
        }
    } else {
        # No title — all rows get empty borders
        for ($r = $Y + 1; $r -lt $Y + $Height - 1; $r++) {
            Move-TuiCursor -Row $r -Col $X
            Write-Host $Script:TUI_BOX_V -NoNewline
            Write-Host (" " * $innerW) -NoNewline
            Write-Host $Script:TUI_BOX_V -NoNewline
        }
    }

    # Bottom border
    Move-TuiCursor -Row ($Y + $Height - 1) -Col $X
    Write-Host "$($Script:TUI_BOX_BL)$hLine$($Script:TUI_BOX_BR)" -NoNewline
}
```

**Section: Key Reading Function**

Implement `Read-TuiKey` — the PowerShell equivalent of `_tui_read_key()`. Uses `[Console]::ReadKey($true)` per D-09 which provides `KeyChar`, `Key`, and `Modifiers` properties:

```powershell
function Read-TuiKey {
    # Sets script-scope variables:
    #   $Script:_tui_rk_result  — symbolic key name (matching TUI_KEY_* constants)
    #   $Script:_tui_rk_digit   — digit character when result is TUI_KEY_NUMBER, else empty string
    # Returns nothing directly. Caller reads the script variables.
    # Per D-09: uses [Console]::ReadKey() — PowerShell's native single-keypress API.

    $Script:_tui_rk_result = $Script:TUI_KEY_UNKNOWN
    $Script:_tui_rk_digit = ''

    # Read key with intercept=true (don't echo to console)
    $keyInfo = $null
    try {
        if ([Console]::KeyAvailable) {
            $keyInfo = [Console]::ReadKey($true)
        } else {
            # Blocking read — ReadKey will wait for input
            $keyInfo = [Console]::ReadKey($true)
        }
    } catch {
        # On error (e.g., input redirected), return unknown
        $Script:_tui_rk_result = $Script:TUI_KEY_UNKNOWN
        return
    }

    $char = $keyInfo.KeyChar
    $key = $keyInfo.Key
    $modifiers = $keyInfo.Modifiers

    # --- Character-based mappings (matching tui.sh lines 205-248) ---

    # Check KeyChar first for printable characters
    switch -Regex ($char.ToString()) {
        "`r" { $Script:_tui_rk_result = $Script:TUI_KEY_ENTER; return }
        "`n" { $Script:_tui_rk_result = $Script:TUI_KEY_ENTER; return }
        ' '  { $Script:_tui_rk_result = $Script:TUI_KEY_SPACE; return }
        '*'  { $Script:_tui_rk_result = $Script:TUI_KEY_ASTERISK; return }
        '-'  { $Script:_tui_rk_result = $Script:TUI_KEY_MINUS; return }
        "`t" { $Script:_tui_rk_result = $Script:TUI_KEY_TAB; return }
        'q'  { $Script:_tui_rk_result = $Script:TUI_KEY_Q; return }
        'Q'  { $Script:_tui_rk_result = $Script:TUI_KEY_Q; return }
        '?'  { $Script:_tui_rk_result = $Script:TUI_KEY_HELP; return }
        'j'  { $Script:_tui_rk_result = $Script:TUI_KEY_DOWN; return }   # vi key: j = down
        'k'  { $Script:_tui_rk_result = $Script:TUI_KEY_UP; return }     # vi key: k = up
        'G'  { $Script:_tui_rk_result = $Script:TUI_KEY_END; return }    # vi key
        'g'  { $Script:_tui_rk_result = $Script:TUI_KEY_HOME; return }   # vi key
        { $_ -match '^\d$' } {
            $Script:_tui_rk_digit = $char.ToString()
            $Script:_tui_rk_result = $Script:TUI_KEY_NUMBER
            return
        }
    }

    # --- ConsoleKey-based mappings (escape sequences handled natively by ReadKey) ---
    # Note: [Console]::ReadKey() already decodes escape sequences.
    # Arrow keys, Home, End, PgUp, PgDn, etc. all map to ConsoleKey enum values.
    # This avoids the manual escape sequence parsing that tui.sh must do.

    switch ($key) {
        'UpArrow'    { $Script:_tui_rk_result = $Script:TUI_KEY_UP; return }
        'DownArrow'  { $Script:_tui_rk_result = $Script:TUI_KEY_DOWN; return }
        'LeftArrow'  { $Script:_tui_rk_result = $Script:TUI_KEY_LEFT; return }
        'RightArrow' { $Script:_tui_rk_result = $Script:TUI_KEY_RIGHT; return }
        'Enter'      { $Script:_tui_rk_result = $Script:TUI_KEY_ENTER; return }
        'Escape'     { $Script:_tui_rk_result = $Script:TUI_KEY_ESC; return }
        'PageUp'     { $Script:_tui_rk_result = $Script:TUI_KEY_PGUP; return }
        'PageDown'   { $Script:_tui_rk_result = $Script:TUI_KEY_PGDN; return }
        'Home'       { $Script:_tui_rk_result = $Script:TUI_KEY_HOME; return }
        'End'        { $Script:_tui_rk_result = $Script:TUI_KEY_END; return }
        'Spacebar'   { $Script:_tui_rk_result = $Script:TUI_KEY_SPACE; return }
        'Tab'        { $Script:_tui_rk_result = $Script:TUI_KEY_TAB; return }
        'Backspace'  { $Script:_tui_rk_result = $Script:TUI_KEY_BACKSPACE; return }
        'Delete'     { $Script:_tui_rk_result = $Script:TUI_KEY_DELETE; return }

        # Ctrl+D handling — ReadKey exposes modifiers
        'D' {
            if ($modifiers -band [System.ConsoleModifiers]::Control) {
                $Script:_tui_rk_result = $Script:TUI_KEY_CTRL_D; return
            }
        }
    }

    # Fallback: process raw byte value for edge cases
    $byteVal = [int][char]$char
    switch ($byteVal) {
        10 { $Script:_tui_rk_result = $Script:TUI_KEY_ENTER; return }
        13 { $Script:_tui_rk_result = $Script:TUI_KEY_ENTER; return }
        27 { $Script:_tui_rk_result = $Script:TUI_KEY_ESC; return }
        127 { $Script:_tui_rk_result = $Script:TUI_KEY_BACKSPACE; return }
```

```powershell
function Show-TuiSelect {
    <#
    .SYNOPSIS
    Single-select TUI menu widget. PowerShell port of tui_select().

    .PARAMETER Title
    Box title displayed in the top border.

    .PARAMETER Subtitle
    Context line displayed below title (e.g., breadcrumb).

    .PARAMETER Items
    Array of item strings to display and choose from.

    .DESCRIPTION
    Renders a full-screen bordered box with items, keyboard navigation,
    scroll indicators, help footer, and number jump.
    Sets $Script:TUI_RESULT to 0-based selected index on Enter.
    Sets $Script:TUI_RESULT to -1 on Esc/q cancel.
    Falls back to numbered text prompt when $_tui_use_tui is $false.

    Matching tui.sh tui_select() behaviors:
      - Items overflow → scroll with ↑more / ↓more indicators
      - PgUp/PgDn jump by page height
      - Home/End jump to first/last
      - Number accumulator: typing 1,2,3 → jumps to item index as digits accumulate
      - Help footer: "↑↓ jk move  ↵ select  Esc/q cancel  ? help  Home End PgUp PgDn"
      - Reverse video highlight on current item ($Script:TUI_REV)
    #>
    param(
        [string]$Title,
        [string]$Subtitle,
        [string[]]$Items
    )

    # Fallback mode:
    if (-not $Script:_tui_use_tui) {
        $result = Show-TuiSelectFallback -Title $Title -Subtitle $Subtitle -Items $Items
        $Script:TUI_RESULT = $result
        return
    }

    Initialize-Tui
    Clear-TuiScreen

    $itemCount = $Items.Count
    $currentIndex = 0
    $topIndex = 0      # first visible item index
    $digitAccum = ''   # multi-digit number accumulator
    $running = $true
    $needsRedraw = $true

    # Get terminal size
    $termRows = try { $Host.UI.RawUI.WindowSize.Height } catch { 24 }
    $termCols = try { $Host.UI.RawUI.WindowSize.Width } catch { 80 }

    # Calculate box dimensions
    $boxWidth = [Math]::Min(76, $termCols - 2)
    if ($boxWidth -lt 20) { $boxWidth = 20 }
    # title row + subtitle row + footer row + 2 border rows = 4 overhead rows
    $visibleRows = [Math]::Max(1, $termRows - 7)
    $boxHeight = $visibleRows + 6

    $boxX = [Math]::Max(0, [Math]::Floor(($termCols - $boxWidth) / 2))
    $boxY = [Math]::Max(0, [Math]::Floor(($termRows - $boxHeight) / 2))
    $innerWidth = $boxWidth - 4  # inside padding

    $scrollIndicator = ''
    $prevNeedsScroll = $false

    while ($running) {
        if ($needsRedraw) {
            Clear-TuiScreen

            # Draw the bordered box with title
            Write-TuiBox -X $boxX -Y $boxY -Width $boxWidth -Height $boxHeight -Title $Title

            # Render subtitle (if provided) below title
            if ($Subtitle) {
                Write-TuiAt -Row ($boxY + 2) -Col ($boxX + 2)
                Write-Host $Subtitle -NoNewline
            }

            # Render visible items
            $needsScroll = $itemCount -gt $visibleRows
            $itemStartRow = $boxY + 3

            # Top scroll indicator
            if ($needsScroll -and $topIndex -gt 0) {
                $scrollIndicator = "↑ $($Script:TUI_DIM)more$($Script:TUI_RESET)"
                Write-TuiAt -Row $itemStartRow -Col ($boxX + 2)
                Write-Host $scrollIndicator -NoNewline
                $itemStartRow++
                $visibleRows--
            }

            for ($i = 0; $i -lt [Math]::Min($visibleRows, $itemCount - $topIndex); $i++) {
                $itemIdx = $topIndex + $i
                $row = $itemStartRow + $i
                $isCurrent = ($itemIdx -eq $currentIndex)

                # Render item at position
                $itemText = $Items[$itemIdx]
                # Truncate to fit inner width
                if ($itemText.Length -gt $innerWidth) {
                    $itemText = $itemText.Substring(0, $innerWidth - 3) + '...'
                }
                $paddedText = $itemText + (' ' * ([Math]::Max(0, $innerWidth - $itemText.Length)))

                Write-TuiAt -Row $row -Col ($boxX + 2)
                if ($isCurrent) {
                    Write-Host "$($Script:TUI_REV)$paddedText$($Script:TUI_RESET)" -NoNewline
                } else {
                    Write-Host $paddedText -NoNewline
                }
            }

            # Bottom scroll indicator
            if ($needsScroll -and ($topIndex + $visibleRows) -lt $itemCount) {
                $bottomRow = $itemStartRow + $visibleRows - 1
                Write-TuiAt -Row $bottomRow -Col ($boxX + 2)
                Write-Host "↓ $($Script:TUI_DIM)more$($Script:TUI_RESET)" -NoNewline
            }

            # Help footer row (above bottom border)
            $footerRow = $boxY + $boxHeight - 2
            $footerText = "↑↓ jk move  ↵ select  Esc/q cancel  ? help  Home End PgUp PgDn"
            Write-TuiAt -Row $footerRow -Col ($boxX + [Math]::Max(0, [Math]::Floor(($boxWidth - $footerText.Length) / 2)))
            Write-Host "$($Script:TUI_DIM)$footerText$($Script:TUI_RESET)" -NoNewline

            $needsRedraw = $false
        }

        # Read key
        Read-TuiKey
        $key = $Script:_tui_rk_result

        switch ($key) {
            $Script:TUI_KEY_UP {
                if ($currentIndex -gt 0) { $currentIndex-- }
                if ($currentIndex -lt $topIndex) { $topIndex = $currentIndex }
                $needsRedraw = $true
            }
            $Script:TUI_KEY_DOWN {
                if ($currentIndex -lt $itemCount - 1) { $currentIndex++ }
                if ($currentIndex -ge $topIndex + $visibleRows) { $topIndex = $currentIndex - $visibleRows + 1 }
                $needsRedraw = $true
            }
            $Script:TUI_KEY_PGUP {
                $currentIndex = [Math]::Max(0, $currentIndex - $visibleRows)
                $topIndex = [Math]::Max(0, $topIndex - $visibleRows)
                if ($currentIndex -lt $topIndex) { $topIndex = $currentIndex }
                $needsRedraw = $true
            }
            $Script:TUI_KEY_PGDN {
                $currentIndex = [Math]::Min($itemCount - 1, $currentIndex + $visibleRows)
                $topIndex = [Math]::Min($itemCount - $visibleRows, $topIndex + $visibleRows)
                if ($topIndex -lt 0) { $topIndex = 0 }
                if ($currentIndex -ge $topIndex + $visibleRows) { $topIndex = $currentIndex - $visibleRows + 1 }
                $needsRedraw = $true
            }
            $Script:TUI_KEY_HOME {
                $currentIndex = 0
                $topIndex = 0
                $needsRedraw = $true
            }
            $Script:TUI_KEY_END {
                $currentIndex = $itemCount - 1
                $topIndex = [Math]::Max(0, $itemCount - $visibleRows)
                $needsRedraw = $true
            }
            $Script:TUI_KEY_ENTER {
                $Script:TUI_RESULT = $currentIndex
                $running = $false
            }
            $Script:TUI_KEY_ESC {
                $Script:TUI_RESULT = -1
                $running = $false
            }
            $Script:TUI_KEY_Q {
                $Script:TUI_RESULT = -1
                $running = $false
            }
            $Script:TUI_KEY_NUMBER {
                # Multi-digit number accumulator (matching tui.sh behavior)
                $digitAccum += $Script:_tui_rk_digit
                $targetIdx = [int]$digitAccum
                if ($targetIdx -ge 1 -and $targetIdx -le $itemCount) {
                    $currentIndex = $targetIdx - 1  # convert 1-based to 0-based
                    if ($currentIndex -lt $topIndex) { $topIndex = $currentIndex }
                    if ($currentIndex -ge $topIndex + $visibleRows) { $topIndex = $currentIndex - $visibleRows + 1 }
                }
                $needsRedraw = $true
                # Reset accumulator after 500ms timeout (simplified: reset on next non-digit key)
            }
            default {
                # Reset digit accumulator on any non-number key
                $digitAccum = ''
            }
        }
    }

    Restore-Tui
}

function Show-TuiSelectFallback {
    <#
    .SYNOPSIS
    Numbered text prompt fallback when TUI is unavailable (matching _tui_select_fallback()).
    #>
    param([string]$Title, [string]$Subtitle, [string[]]$Items)

    Write-Host "$Title"
    if ($Subtitle) { Write-Host $Subtitle }
    for ($i = 0; $i -lt $Items.Count; $i++) {
        $num = $i + 1
        Write-Host "  $num) $($Items[$i])"
    }
    Write-Host ""
    $choice = Read-Host "Enter number (1-$($Items.Count)) or 0 to cancel"
    $num = 0
    if ([int]::TryParse($choice, [ref]$num)) {
        if ($num -ge 1 -and $num -le $Items.Count) {
            return $num - 1
        }
    }
    return -1
}
```
