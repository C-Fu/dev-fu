    **Template pattern for .ps1 module scripts (must follow for all files):**

    ```powershell
    # @name: Install {Tool Name}
    # @params:
    # @platforms: windows
    # @version: 1.0.0
    # @deps:
    # @timeout: {seconds}

    $ErrorActionPreference = 'Stop'

    [Console]::OutputEncoding = [System.Text.Encoding]::UTF8

    # Idempotent guard
    if (Get-Command {tool_exe} -ErrorAction SilentlyContinue) {
        Write-Host "{Name} already installed: $({tool_exe} --version 2>$null | Select-Object -First 1)"
        exit 0
    }

    # Platform dispatch via FLU_PKG_MGR
    switch ($env:FLU_PKG_MGR) {
        'winget' {
            winget install --id {winget_id} --silent --accept-package-agreements
        }
        'choco' {
            choco install {choco_pkg} -y
        }
        'scoop' {
            scoop install {scoop_pkg}
        }
        default {
            Write-Host "{Name} is not available through any supported Windows package manager."
            Write-Host "Visit {website} for manual installation instructions."
            exit 1
        }
    }

    if ($LASTEXITCODE -eq 0) {
        Write-Host "{Name} installed successfully"
    }
    exit $LASTEXITCODE
    ```

    **For tools not available via winget/choco/scoop** (direct download tools like lazygit), use this pattern:

    ```powershell
    # @name: Install lazygit
    # @params:
    # @platforms: windows
    # @version: 1.0.0
    # @deps: curl,wget
    # @timeout: 600

    $ErrorActionPreference = 'Stop'

    [Console]::OutputEncoding = [System.Text.Encoding]::UTF8

    if (Get-Command lazygit -ErrorAction SilentlyContinue) {
        Write-Host "lazygit already installed"
        exit 0
    }

    Write-Host "Installing lazygit..."

    # Detect architecture
    $arch = if ($env:PROCESSOR_ARCHITECTURE -match 'ARM64') { 'arm64' } else { 'x86_64' }

    # Download latest release
    $url = "https://github.com/jesseduffield/lazygit/releases/latest/download/lazygit_${env:FLU_OS}_${arch}.zip"
    $zipPath = "$env:TEMP\lazygit.zip"

    Invoke-WebRequest -Uri $url -OutFile $zipPath -UseBasicParsing
    Expand-Archive -Path $zipPath -DestinationPath "$env:TEMP\lazygit" -Force
    Move-Item "$env:TEMP\lazygit\lazygit.exe" "$env:ProgramFiles\lazygit\lazygit.exe" -Force
    Remove-Item $zipPath, "$env:TEMP\lazygit" -Recurse -Force

    Write-Host "lazygit installed successfully"
    ```

    **For remove operations:**

    ```powershell
    # @name: Remove lazygit
    # @params:
    # @platforms: windows
    # @version: 1.0.0
    # @deps:
    # @timeout: 120

    $ErrorActionPreference = 'Stop'

    if (-not (Get-Command lazygit -ErrorAction SilentlyContinue)) {
        Write-Host "lazygit is not installed"
        exit 0
    }

    switch ($env:FLU_PKG_MGR) {
        'winget' { winget uninstall --id JesseDuffield.lazygit --silent }
        'choco'  { choco uninstall lazygit -y }
        'scoop'  { scoop uninstall lazygit }
        default  { Remove-Item "$env:ProgramFiles\lazygit\lazygit.exe" -Force -ErrorAction SilentlyContinue }
    }

    Write-Host "lazygit removed successfully"
    ```
