**Main event loop in tui_select():**
```
while :; do
  _tui_render_select
  key=$(_tui_read_key)
  
  case "$key" in
    $TUI_KEY_UP)
      # ENGN-01: Up arrow / k vi key
      if [ "$_ts_cursor" -gt 1 ]; then
        _ts_cursor=$((_ts_cursor - 1))
        # Adjust scroll if cursor scrolled above viewport
        if [ "$_ts_cursor" -lt "$_ts_scroll" ]; then
          _ts_scroll=$((_ts_scroll - 1))
        fi
      fi
      _ts_go_digits=''
      _ts_error_msg=''
      ;;
    $TUI_KEY_DOWN)
      # ENGN-01: Down arrow / j vi key
      if [ "$_ts_cursor" -lt "$_ts_count" ]; then
        _ts_cursor=$((_ts_cursor + 1))
        # Adjust scroll if cursor went below viewport
        _ts_bottom=$((_ts_scroll + _ts_page_size - 1))
        if [ "$_ts_cursor" -gt "$_ts_bottom" ]; then
          _ts_scroll=$((_ts_scroll + 1))
        fi
      fi
      _ts_go_digits=''
      _ts_error_msg=''
      ;;
    $TUI_KEY_PGUP)
      # ENGN-05: Page Up
      _ts_scroll=$((_ts_scroll - _ts_page_size))
      [ "$_ts_scroll" -lt 1 ] && _ts_scroll=1
      _ts_cursor=$_ts_scroll
      _ts_go_digits=''
      _ts_error_msg=''
      ;;
    $TUI_KEY_PGDN)
      # ENGN-05: Page Down
      _ts_scroll=$((_ts_scroll + _ts_page_size))
      _ts_max_scroll=$((_ts_count - _ts_page_size + 1))
      [ "$_ts_max_scroll" -lt 1 ] && _ts_max_scroll=1
      [ "$_ts_scroll" -gt "$_ts_max_scroll" ] && _ts_scroll=$_ts_max_scroll
      _ts_bottom=$((_ts_scroll + _ts_page_size - 1))
      [ "$_ts_bottom" -gt "$_ts_count" ] && _ts_bottom=$_ts_count
      _ts_cursor=$_ts_bottom
      _ts_go_digits=''
      _ts_error_msg=''
      ;;
    $TUI_KEY_HOME)
      # ENGN-06: Home / g vi key
      _ts_cursor=1
      _ts_scroll=1
      _ts_go_digits=''
      _ts_error_msg=''
      ;;
    $TUI_KEY_END)
      # ENGN-06: End / G vi key
      _ts_cursor=$_ts_count
      _ts_max_scroll=$((_ts_count - _ts_page_size + 1))
      [ "$_ts_max_scroll" -lt 1 ] && _ts_max_scroll=1
      _ts_scroll=$_ts_max_scroll
      _ts_go_digits=''
      _ts_error_msg=''
      ;;
    $TUI_KEY_ENTER)
      # ENGN-02: Select with Enter
      tui_restore
      index=$((_ts_cursor - 1))  # Convert to 0-based
      TUI_RESULT=$index
      printf '%d\n' "$index"
      return 0
      ;;
    $TUI_KEY_ESC|$TUI_KEY_Q)
      # ENGN-02: Cancel with Esc or q
      tui_restore
      return 1
      ;;
    $TUI_KEY_HELP)
      # D-16: Toggle full help display
      if [ "$_ts_show_help" = true ]; then
        _ts_show_help=false
      else
        _ts_show_help=true
      fi
      ;;
    $TUI_KEY_NUMBER)
      # ENGN-07 / D-20: Number jump accumulator
      _ts_go_digits="${_ts_go_digits}${_tui_digit_char}"
      # If accumulated number >= count, we might have a valid multi-digit prefix
      # Check if accumulated value is a valid item number
      _ts_target=0
      # Calculate numeric value from digit string (POSIX: no $((10#$d)) syntax)
      _ts_target=$(printf '%s' "$_ts_go_digits" | awk '{print $1 + 0}')
      if [ "$_ts_target" -ge 1 ] && [ "$_ts_target" -le "$_ts_count" ]; then
        # Valid number — check if this could be a prefix of a larger number
        _ts_next=$((_ts_target * 10))
        if [ "$_ts_next" -gt "$_ts_count" ] && [ "$_ts_next" -le $((_ts_count * 10)) ]; then
          # No valid extension possible — jump immediately
          _ts_cursor=$_ts_target
          _ts_scroll=$((_ts_target - _ts_page_size / 2))
          [ "$_ts_scroll" -lt 1 ] && _ts_scroll=1
          _ts_go_digits=''
        fi
        # Otherwise, wait for timeout or next digit
      elif [ "$_ts_target" -gt "$_ts_count" ]; then
        # D-22: Out of range — flash error
        _ts_error_msg="Item $_ts_target not found"
        _ts_go_digits=''
      fi
      ;;
    $TUI_KEY_UNKNOWN)
      # Ignore unrecognized keys
      ;;
  esac
done
```

**Number jump timeout (per D-20):**
The auto-timeout is handled by `_tui_read_key()` which uses `TUI_KEY_TIMEOUT` (deciseconds). After reading a number digit, if the next keypress is NOT another number digit (i.e., the user paused longer than the timeout or pressed a non-digit key), the accumulated number should be executed:
- After processing any non-NUMBER key, check if `_ts_go_digits` is non-empty
- If so, jump to the accumulated number:
  ```
  _ts_target=$(printf '%s' "$_ts_go_digits" | awk '{print $1 + 0}')
  if [ "$_ts_target" -ge 1 ] && [ "$_ts_target" -le "$_ts_count" ]; then
    _ts_cursor=$_ts_target
    # Center the viewport on the target
    _ts_scroll=$((_ts_target - _ts_page_size / 2))
    [ "$_ts_scroll" -lt 1 ] && _ts_scroll=1
    _ts_max_scroll=$((_ts_count - _ts_page_size + 1))
    [ "$_ts_max_scroll" -lt 1 ] && _ts_max_scroll=1
    [ "$_ts_scroll" -gt "$_ts_max_scroll" ] && _ts_scroll=$_ts_max_scroll
  else
    _ts_error_msg="Item $_ts_target not found"
  fi
  _ts_go_digits=''
  ```
- This timeout-flush check goes at the TOP of the main loop, before rendering — check `_ts_go_digits` and if it's stale (set > TUI_KEY_TIMEOUT ago), flush it.
- Simpler approach: on any non-digit key, if `_ts_go_digits` is non-empty, execute the jump. This is reliable because `_tui_read_key()` blocks until a key is pressed — the gap between digit and next key IS the user's think time.
